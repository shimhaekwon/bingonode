### M4 — bingo 대량 upsert 트랜잭션화

`models/bingoModel.js` — `setUpsertMany()` 신규 추가 (stockModel.upsertMany와 동일 패턴):
```js
async function setUpsertMany(rows) {
  await ensureReady();
  if (!Array.isArray(rows) || rows.length === 0) return { written: 0 };
  let written = 0;
  await execRun('BEGIN TRANSACTION');
  try {
    for (const r of rows) {
      if (!r || !Number.isInteger(r.seq)) continue;
      const params = [r.seq, r.no1, r.no2, r.no3, r.no4, r.no5, r.no6, r.no7 ?? null];
      const result = await execRun(BingoQueries.setUpsert, params);
      if (result.changes > 0) written++;
    }
    await execRun('COMMIT');
  } catch (err) {
    await execRun('ROLLBACK').catch(() => {});
    throw err;
  }
  return { written };
}
```

`services/bingoService.js` — 누락회차 블록 + verify-latest 블록의 행단위 upsert 루프(존재 여부 분기 포함)를 `setUpsertMany()` 단일 호출로 교체.

- 기존 단건 `setUpsert(seq,row)`는 보존(컨트롤러 `setUpsert`가 사용 — 더함없이/다름없이).
- `r.no7 ?? null` — undefined 보너스번호를 null로 정규화 (틀림없이).
